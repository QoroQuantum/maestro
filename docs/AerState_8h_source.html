173 void Configure(const char *key, const char *value) override {
174 if (std::string("method") == key) {
175 if (std::string("statevector") == value)
176 simulationType = SimulationType::kStatevector;
177 else if (std::string("matrix_product_state") == value)
178 simulationType = SimulationType::kMatrixProductState;
179 else if (std::string("stabilizer") == value)
180 simulationType = SimulationType::kStabilizer;
181 else if (std::string("tensor_network") == value)
182 simulationType = SimulationType::kTensorNetwork;
183 else
184 simulationType = SimulationType::kOther;
185 } else if (std::string("matrix_product_state_truncation_threshold") ==
186 key) {
187 singularValueThreshold = std::stod(value);
188 if (singularValueThreshold > 0.)
189 limitEntanglement = true;
190 else
191 limitEntanglement = false;
192 } else if (std::string("matrix_product_state_max_bond_dimension") == key) {
193 chi = std::stoi(value);
194 if (chi > 0)
195 limitSize = true;
196 else
197 limitSize = false;
198 } else if (std::string("mps_sample_measure_algorithm") == key)
199 useMPSMeasureNoCollapse = std::string("mps_probabilities") == value;
200
201 state->configure(key, value);

405 double ExpectationValue(const std::string &pauliStringOrig) override {
406 if (pauliStringOrig.empty()) return 1.0;
407
408 std::string pauliString = pauliStringOrig;
409 if (pauliString.size() > GetNumberOfQubits()) {
410 for (size_t i = GetNumberOfQubits(); i < pauliString.size(); ++i) {
411 const auto pauliOp = toupper(pauliString[i]);
412 if (pauliOp != 'I' && pauliOp != 'Z') return 0.0;
413 }
414
415 pauliString.resize(GetNumberOfQubits());
416 }
417
418 AER::reg_t qubits;
419 std::string pauli;
420
421 pauli.reserve(pauliString.size());
422 qubits.reserve(pauliString.size());
423
424 for (size_t q = 0; q < pauliString.size(); ++q) {
425 const char p = toupper(pauliString[q]);
426 if (p == 'I') continue;
427
428 pauli.push_back(p);
429 qubits.push_back(q);
430 }
431
432 if (qubits.empty()) return 1.0;
433
434 // qiskit aer expects the pauli string in reverse order
435 std::reverse(pauli.begin(), pauli.end());
436
437 return state->expval_pauli(qubits, pauli);
438 }

505 void SaveState() override {
506 if (!state) return;
507
508 const auto numQubits = GetNumberOfQubits();
509
510 if (simulationType == SimulationType::kStatevector) {
511 SaveStateToInternalDestructive();
512 state->initialize_statevector(numQubits, savedAmplitudes.data(), true);
513 return;
514 }
515
516 bool saved = false;
517
518 if (state->is_initialized()) {
519 AER::Operations::Op op;
520
521 op.type = AER::Operations::OpType::save_state;
522 op.name = "save_state";
523 op.save_type = AER::Operations::DataSubType::single;
524 op.string_params.push_back("s");
525
526 for (size_t q = 0; q < numQubits; ++q) op.qubits.push_back(q);
527
528 state->buffer_op(std::move(op));
529 Flush();
530
531 // get the state from the last result
532 AER::ExperimentResult &last_result = state->last_result();
533 // state should be in last_result.data
534 if (last_result.status == AER::ExperimentResult::Status::completed) {
535 savedState = std::move(last_result.data);
536 saved = true;
537 }
538 } else {
539 // try get the state from the last result
540 AER::ExperimentResult &last_result_prev = state->last_result();
541 // state should be in last_result.data
542 if (last_result_prev.status == AER::ExperimentResult::Status::completed) {
543 savedState = std::move(last_result_prev.data);
544 saved = true;
545 }
546 }
547
548 // this is a hack, for statevector and matrix product state if the last op
549 // is executed, it can destroy the state! see also the workaround for
550 // statevector for the stabilizer at least for now it doesn't seem to do
551 // that
552 // TODO: check everything!!!!
553 if (saved && simulationType == SimulationType::kMatrixProductState)
554 RestoreState();
555 }

724 Types::qubit_t MeasureNoCollapse() override {
725 if (simulationType == SimulationType::kStatevector) {
726 const double prob =
727 1. - uniformZeroOne(rng); // this excludes 0 as probabiliy
728 double accum = 0;
729 Types::qubit_t state = 0;
730 for (Types::qubit_t i = 0; i < savedAmplitudes.size(); ++i) {
731 accum += std::norm(savedAmplitudes[i]);
732 if (prob <= accum) {
733 state = i;
734 break;
735 }
736 }
737
738 return state;
739 }
740
741 throw std::runtime_error(
742 "AerState::MeasureNoCollapse: Invalid simulation type for measuring "
743 "all the qubits without collapsing the state.");
744
745 return 0;
746 }
