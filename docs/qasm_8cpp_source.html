64 void GenerateCircuit(int nrGates, int nrQubits, double probReset = 0.,
65 double probMeasurement = 0.) {
66 std::random_device rd;
67 std::mt19937 g(rd());
68
69 auto dblGen = bdata::random(-2. * M_PI, 2. * M_PI);
70 auto dblGenIter = dblGen.begin();
71
72 auto gateGen = bdata::random(
73 0, static_cast<int>(Circuits::QuantumGateType::kCUGateType));
74 auto gateGenIter = gateGen.begin();
75
76 auto typeGen = bdata::random();
77 auto typeGenIter = typeGen.begin();
78
79 const double cumProb = probReset + probMeasurement;
80
81 // TODO: Maybe insert from time to time a random number generating 'gate'
82 // and a conditioned random one, those should not affect results?
83 for (int gateNr = 0; gateNr < nrGates; ++gateNr, ++gateGenIter) {
84 const double typeOpProb = *typeGenIter;
85 ++typeGenIter;
86
87 Types::qubits_vector qubits(nrQubits);
88 std::iota(qubits.begin(), qubits.end(), 0);
89 std::shuffle(qubits.begin(), qubits.end(), g);
90
91 if (typeOpProb < probReset) {
92 // reset operation
93 auto resetOp = Circuits::CircuitFactory<>::CreateReset({qubits[0]});
94 randomCirc->AddOperation(resetOp);
95 continue;
96 } else if (typeOpProb < cumProb) {
97 // measurement operation
98 auto measOp = Circuits::CircuitFactory<>::CreateMeasurement(
99 {{qubits[0], qubits[0]}});
100 randomCirc->AddOperation(measOp);
101 continue;
102 }
103
104 // create a random gate and add it to the circuit
105
106 // first, pick randomly three qubits (depending on the randomly chosen
107 // gate type, not all of them will be used)
108 auto q1 = qubits[0];
109 auto q2 = qubits[1];
110 auto q3 = qubits[2];
111
112 // now some random parameters, again, they might be ignored
113 const double param1 = *dblGenIter;
114 ++dblGenIter;
115 const double param2 = *dblGenIter;
116 ++dblGenIter;
117 const double param3 = *dblGenIter;
118 ++dblGenIter;
119
120 const Circuits::QuantumGateType gateType =
121 static_cast<Circuits::QuantumGateType>(*gateGenIter);
122
123 auto theGate = Circuits::CircuitFactory<>::CreateGate(
124 gateType, q1, q2, q3, param1, param2, param3);
125 randomCirc->AddOperation(theGate);
126 }
127 }
