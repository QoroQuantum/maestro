71 void CreateNetwork(const std::vector<Types::qubit_t> &qubits,
72 const std::vector<size_t> &cbits) {
73 size_t qubitsOffset = 0;
74 size_t cbitsOffset = 0;
75
76 for (size_t i = 0; i < qubits.size(); ++i) {
77 const size_t numQubits = qubits[i];
78 const size_t numBits = (i < cbits.size() ? cbits[i] : 0);
79 hosts.emplace_back(std::make_shared<SimpleHost<Time>>(
80 i, qubitsOffset, numQubits, cbitsOffset, numBits));
81 qubitsOffset += numQubits;
82 cbitsOffset += numBits;
83 }
84
85 for (size_t i = 0; i < hosts.size(); ++i) {
86 std::static_pointer_cast<SimpleHost<Time>>(hosts[i])->SetEntangledQubitId(
87 qubitsOffset);
88 std::static_pointer_cast<SimpleHost<Time>>(hosts[i])
89 ->SetEntangledQubitMeasurementBit(cbitsOffset);
90 ++qubitsOffset;
91 ++cbitsOffset;
92 }
93
94 controller = std::make_shared<Controller>();
95 }

361 ExecuteResults RepeatedExecute(
362 const std::shared_ptr<Circuits::Circuit<Time>> &circuit,
363 size_t shots = 1000) override {
364 if (!controller || !circuit) return {};
365
366 distCirc = controller->DistributeCircuit(BaseClass::getptr(), circuit);
367
368#ifdef _DEBUG
369 for (auto q : distCirc->AffectedQubits()) {
370 if (q >= GetNumQubits()) {
371 std::cout
372 << "This is a distributed circuit, using entanglement or cutting"
373 << std::endl;
374 break;
375 }
376 }
377#endif
378
379 if (!simulator || distCirc->empty()) return {};
380
381 auto simType = simulator->GetType();
382 if (distCirc->HasOpsAfterMeasurements() &&
383 (
384#ifndef NO_QISKIT_AER
385 simType == Simulators::SimulatorType::kCompositeQiskitAer ||
386#endif
387 simType == Simulators::SimulatorType::kCompositeQCSim))
388 distCirc->MoveMeasurementsAndResets();
389
390 auto method = simulator->GetSimulationType();
391
392 const auto saveSimType = simType;
393 const auto saveMethod = method;
394
395 if (GetOptimizeSimulator() && distCirc->IsClifford() &&
396 method != Simulators::SimulationType::kStabilizer
397 // this is for the gpu simulator, as it doesn't support stabilizer
398#ifdef __linux__
399 && simType != Simulators::SimulatorType::kGpuSim
400#endif
401 ) {
402 method = Simulators::SimulationType::kStabilizer;
403
404 if (simType == Simulators::SimulatorType::kCompositeQCSim)
405 simType = Simulators::SimulatorType::kQCSim;
406#ifndef NO_QISKIT_AER
407 else if (simType == Simulators::SimulatorType::kCompositeQiskitAer)
408 simType = Simulators::SimulatorType::kQiskitAer;
409#endif
410 }
411
412 ExecuteResults res;
413 const size_t nrQubits = GetNumQubits() + GetNumNetworkEntangledQubits();
414 const size_t nrCbitsResults = GetNumClassicalBits();
415
416 maxBondDim =
417 simulator->GetConfiguration("matrix_product_state_max_bond_dimension");
418 singularValueThreshold = simulator->GetConfiguration(
419 "matrix_product_state_truncation_threshold");
420 mpsSample = simulator->GetConfiguration("mps_sample_measure_algorithm");
421
422 // do that only if the optimization for simulator is on and the estimator is
423 // available, ortherwise an 'optimal' simulator won't be created
424 if (optimizeSimulator && simulatorsEstimator) {
425 simulator->Clear();
426 GetState().Clear();
427 }
428
429 std::vector<bool> executed;
430 auto optSim =
431 ChooseBestSimulator(distCirc, shots, nrQubits, nrQubits, nrCbitsResults,
432 simType, method, executed);
433
434 lastSimulatorType = simType;
435 lastMethod = method;
436
437 size_t nrThreads = GetMaxSimulators();
438
439#ifdef __linux__
440 if (simType == Simulators::SimulatorType::kGpuSim) nrThreads = 1;
441#endif
442 if (method == Simulators::SimulationType::kStatevector &&
443 !distCirc->HasOpsAfterMeasurements())
444 nrThreads = 1;
445
446 nrThreads = std::min(nrThreads, std::max<size_t>(shots, 1ULL));
447
448 std::mutex resultsMutex;
449
450 const auto dcirc = distCirc;
451
452 if (nrThreads > 1) {
453 // since it's going to execute on multiple threads, free the memory from
454 // the network's simulator and state, it's going to use other ones,
455 // created in the threads if optimization already exists, it will be
456 // cloned in the threads, otherwise a new one will be created in the
457 // threads
458 if (!optimizeSimulator ||
459 !simulatorsEstimator) // otherwise it was already cleared
460 {
461 simulator->Clear();
462 GetState().Clear();
463 }
464
465 const size_t cntPerThread = std::max<size_t>(shots / nrThreads, 1ULL);
466
467 threadsPool.Resize(nrThreads);
468 threadsPool.SetFinishLimit(shots);
469
470 while (shots > 0) {
471 const size_t curCnt = std::min(cntPerThread, shots);
472
473 shots -= curCnt;
474
475 auto job = std::make_shared<ExecuteJob<Time>>(
476 dcirc, res, curCnt, nrQubits, nrQubits, nrCbitsResults, simType,
477 method, resultsMutex);
478 job->optimiseMultipleShotsExecution = GetOptimizeSimulator();
479
480 job->maxBondDim = maxBondDim;
481 job->mpsSample = mpsSample;
482 job->singularValueThreshold = singularValueThreshold;
483
484 if (optSim) {
485 job->optSim = optSim->Clone();
486 job->executedGates = executed;
487 }
488
489 threadsPool.AddRunJob(std::move(job));
490 }
491
492 threadsPool.WaitForFinish();
493 threadsPool.Stop();
494 } else {
495 const size_t curCnt = shots;
496
497 auto job = std::make_shared<ExecuteJob<Time>>(
498 dcirc, res, curCnt, nrQubits, nrQubits, nrCbitsResults, simType,
499 method, resultsMutex);
500 job->optimiseMultipleShotsExecution = GetOptimizeSimulator();
501
502 job->maxBondDim = maxBondDim;
503 job->mpsSample = mpsSample;
504 job->singularValueThreshold = singularValueThreshold;
505
506 if (optSim) {
507 optSim->SetMultithreading(true);
508 job->optSim = optSim;
509 job->executedGates = executed;
510 } else {
511 if (simulator && method == saveMethod && simType == saveSimType) {
512 optSim = simulator;
513 job->optSim = optSim;
514 simulator = nullptr;
515 }
516 }
517
518 job->DoWorkNoLock();
519 if (!recreateIfNeeded) simulator = job->optSim;
520 }
521
522 if (recreateIfNeeded) CreateSimulator(saveSimType, saveMethod);
523
524 ConvertBackResults(res);
525
526 return res;
527 }
