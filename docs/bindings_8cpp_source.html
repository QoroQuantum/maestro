119 // Parse the QASM circuit first to determine the number of qubits

122 if (parser.Failed() || !circuit) {

125
126 // Get the number of qubits from the circuit
127 int num_qubits = static_cast<int>(circuit->GetMaxQubitIndex()) + 1;
128
129 // Create a simple simulator with the correct number of qubits
130 unsigned long int sim_handle = CreateSimpleSimulator(num_qubits);
131 if (sim_handle == 0) {
132 return nb::none();
133 }
134
135 // Add optimization simulator with the desired types
136 int result = RemoveAllOptimizationSimulatorsAndAdd(
137 sim_handle, static_cast<int>(sim_type),

141 DestroySimpleSimulator(sim_handle);
142 return nb::none();

147 config["shots"] = shots;
148
149 // --- NEW CONFIGURATION VARIABLES ---
150 if (max_bond_dimension.has_value()) {

152 *max_bond_dimension;
153 }
154
155 if (singular_value_threshold.has_value()) {
156 config["matrix_product_state_truncation_threshold"] =

163 // Execute the circuit

167 // Clean up simulator
168 DestroySimpleSimulator(sim_handle);
169

172 }
173
174 // Parse JSON result
175 try {
176 boost::json::value json_result = boost::json::parse(result_str);

182
