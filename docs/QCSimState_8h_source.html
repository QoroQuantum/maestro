220 void Configure(const char *key, const char *value) override {
221 if (std::string("method") == key) {
222 if (std::string("statevector") == value)

224 else if (std::string("matrix_product_state") == value)

244 } else
245 limitSize = false;
246 } else if (std::string("mps_sample_measure_algorithm") == key)
247 useMPSMeasureNoCollapse = std::string("mps_probabilities") == value;

335 size_t Measure(const Types::qubits_vector &qubits) override {
336 // TODO: this is inefficient, maybe implement it better in qcsim
337 // for now it has the possibility of measuring a qubits interval, but not a
338 // list of qubits
339
340 size_t res = 0;
341 size_t mask = 1ULL;
342
343 DontNotify();
344 if (simulationType == SimulationType::kStatevector) {
345 for (size_t qubit : qubits) {
346 if (state->MeasureQubit(static_cast<unsigned int>(qubit))) res |= mask;
347 mask <<= 1;
348 }
349 } else if (simulationType == SimulationType::kStabilizer) {
350 for (size_t qubit : qubits) {
351 if (cliffordSimulator->MeasureQubit(static_cast<unsigned int>(qubit)))
352 res |= mask;
353 mask <<= 1;
354 }
355 } else if (simulationType == SimulationType::kTensorNetwork) {
356 for (size_t qubit : qubits) {
357 if (tensorNetwork->Measure(static_cast<unsigned int>(qubit)))
358 res |= mask;
359 mask <<= 1;
360 }
361 } else {
362 /*
363 for (size_t qubit : qubits)
364 {
365 if (mpsSimulator->MeasureQubit(static_cast<unsigned int>(qubit)))
366 res |= mask;
367 mask <<= 1;
368 }
369 */
370 const std::set<Eigen::Index> qubitsSet(qubits.begin(), qubits.end());
371 auto measured = mpsSimulator->MeasureQubits(qubitsSet);
372 for (Types::qubit_t qubit : qubits) {
373 if (measured[qubit]) res |= mask;
374 mask <<= 1;
375 }
376 }
377 Notify();
378
379 NotifyObservers(qubits);
380
381 return res;

390 void ApplyReset(const Types::qubits_vector &qubits) override {
391 QC::Gates::PauliXGate xGate;
392
393 DontNotify();
394 if (simulationType == SimulationType::kStatevector) {
395 for (size_t qubit : qubits)
396 if (state->MeasureQubit(static_cast<unsigned int>(qubit)))
397 state->ApplyGate(xGate, static_cast<unsigned int>(qubit));
398 } else if (simulationType == SimulationType::kStabilizer) {
399 for (size_t qubit : qubits)
400 if (cliffordSimulator->MeasureQubit(static_cast<unsigned int>(qubit)))
401 cliffordSimulator->ApplyX(static_cast<unsigned int>(qubit));
402 } else if (simulationType == SimulationType::kTensorNetwork) {
403 for (size_t qubit : qubits)
404 if (tensorNetwork->Measure(static_cast<unsigned int>(qubit)))
405 tensorNetwork->AddGate(xGate, static_cast<unsigned int>(qubit));
406 } else {
407 for (size_t qubit : qubits)
408 if (mpsSimulator->MeasureQubit(static_cast<unsigned int>(qubit)))
409 mpsSimulator->ApplyGate(xGate, static_cast<unsigned int>(qubit));
410 }
411 Notify();
412
413 NotifyObservers(qubits);

549 std::unordered_map<Types::qubit_t, Types::qubit_t> SampleCounts(
550 const Types::qubits_vector &qubits, size_t shots = 1000) override {
551 if (qubits.empty() || shots == 0) return {};
552
553 // TODO: this is inefficient, maybe implement it better in qcsim
554 // for now it has the possibility of measuring a qubits interval, but not a
555 // list of qubits
556 std::unordered_map<Types::qubit_t, Types::qubit_t> result;
557
558 DontNotify();
559
560 if (simulationType == SimulationType::kMatrixProductState) {
561 bool normal = true;
562 if (useMPSMeasureNoCollapse) {
563 // check to see if it can be used
564 std::unordered_set qset(qubits.begin(), qubits.end());
565 if (qset.size() == GetNumberOfQubits()) {
566 // it can!
567 normal = false;
568 for (size_t shot = 0; shot < shots; ++shot) {
569 const size_t measRaw = MeasureNoCollapse();
570 size_t meas = 0;
571 size_t mask = 1ULL;
572
573 // translate the measurement
574 for (auto q : qubits) {
575 const size_t qubitMask = 1ULL << q;
576 if (measRaw & qubitMask) meas |= mask;
577 mask <<= 1ULL;
578 }
579
580 ++result[meas];
581 }
582 }
583 }
584
585 if (normal) {
586 auto savedState = mpsSimulator->getState();
587 for (size_t shot = 0; shot < shots; ++shot) {
588 const size_t meas = Measure(qubits);
589 ++result[meas];
590 mpsSimulator->setState(savedState);
591 }
592 }
593 } else if (simulationType == SimulationType::kStabilizer) {
594 cliffordSimulator->SaveState();
595 for (size_t shot = 0; shot < shots; ++shot) {
596 const size_t meas = Measure(qubits);
597 ++result[meas];
598 cliffordSimulator->RestoreState();
599 }
600 cliffordSimulator->ClearSavedState();
601 } else if (simulationType == SimulationType::kTensorNetwork) {
602 tensorNetwork->SaveState();
603 for (size_t shot = 0; shot < shots; ++shot) {
604 const size_t meas = Measure(qubits);
605 ++result[meas];
606 tensorNetwork->RestoreState();
607 }
608 tensorNetwork->ClearSavedState();
609 } else {
610 if (shots > 1) {
611 const auto &statev = state->getRegisterStorage();
612
613 const Utils::Alias alias(statev);
614
615 for (size_t shot = 0; shot < shots; ++shot) {
616 const double prob = 1. - uniformZeroOne(rng);
617 const size_t measRaw = alias.Sample(prob);
618
619 size_t meas = 0;
620 size_t mask = 1ULL;
621 for (auto q : qubits) {
622 const size_t qubitMask = 1ULL << q;
623 if ((measRaw & qubitMask) != 0) meas |= mask;
624 mask <<= 1ULL;
625 }
626
627 ++result[meas];
628 }
629 } else {
630 for (size_t shot = 0; shot < shots; ++shot) {
631 const size_t measRaw = MeasureNoCollapse();
632 size_t meas = 0;
633 size_t mask = 1ULL;
634
635 for (auto q : qubits) {
636 const size_t qubitMask = 1ULL << q;
637 if ((measRaw & qubitMask) != 0) meas |= mask;
638 mask <<= 1ULL;
639 }
640
641 ++result[meas];
642 }
643 }
644 }
645
646 Notify();
647 NotifyObservers(qubits);
648
649 return result;
650 }

663 double ExpectationValue(const std::string &pauliStringOrig) override {
664 if (pauliStringOrig.empty()) return 1.0;
665
666 std::string pauliString = pauliStringOrig;
667 if (pauliString.size() > GetNumberOfQubits()) {
668 for (size_t i = GetNumberOfQubits(); i < pauliString.size(); ++i) {
669 const auto pauliOp = toupper(pauliString[i]);
670 if (pauliOp != 'I' && pauliOp != 'Z') return 0.0;
671 }
672
673 pauliString.resize(GetNumberOfQubits());
674 }
675
676 if (simulationType == SimulationType::kStabilizer)
677 return cliffordSimulator->ExpectationValue(pauliString);
678 else if (simulationType == SimulationType::kTensorNetwork)
679 return tensorNetwork->ExpectationValue(pauliString);
680
681 // statevector or mps
682 static const QC::Gates::PauliXGate<> xgate;
683 static const QC::Gates::PauliYGate<> ygate;
684 static const QC::Gates::PauliZGate<> zgate;
685
686 std::vector<QC::Gates::AppliedGate<Eigen::MatrixXcd>> pauliStringVec;
687 pauliStringVec.reserve(pauliString.size());
688
689 for (size_t q = 0; q < pauliString.size(); ++q) {
690 switch (toupper(pauliString[q])) {
691 case 'X': {
692 QC::Gates::AppliedGate<Eigen::MatrixXcd> ag(
693 xgate.getRawOperatorMatrix(), static_cast<Types::qubit_t>(q));
694 pauliStringVec.emplace_back(std::move(ag));
695 } break;
696 case 'Y': {
697 QC::Gates::AppliedGate<Eigen::MatrixXcd> ag(
698 ygate.getRawOperatorMatrix(), static_cast<Types::qubit_t>(q));
699 pauliStringVec.emplace_back(std::move(ag));
700 } break;
701 case 'Z': {
702 QC::Gates::AppliedGate<Eigen::MatrixXcd> ag(
703 zgate.getRawOperatorMatrix(), static_cast<Types::qubit_t>(q));
704 pauliStringVec.emplace_back(std::move(ag));
705 } break;
706 case 'I':
707 [[fallthrough]];
708 default:
709 break;
710 }
711 }
712
713 if (pauliStringVec.empty()) return 1.0;
714
715 if (simulationType == SimulationType::kMatrixProductState)
716 return mpsSimulator->ExpectationValue(pauliStringVec).real();
717
718 return state->ExpectationValue(pauliStringVec).real();
719 }
