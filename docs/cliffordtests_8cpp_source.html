52 // fills randomly the circuit with gates
53 void GenerateCircuit(int nrGates) {
54 std::random_device rd;
55 std::mt19937 g(rd());
56
57 auto dblGen = bdata::random(-2. * M_PI, 2. * M_PI);
58 auto dblGenIter = dblGen.begin();
59
60 auto gateGen = bdata::random(
61 0, static_cast<int>(Circuits::QuantumGateType::kCZGateType));
62 auto gateGenIter = gateGen.begin();
63
64 // TODO: Maybe insert from time to time a random number generating 'gate'
65 // and a conditioned random one, those should not affect results?
66 for (int gateNr = 0; gateNr < nrGates; ++gateNr, ++gateGenIter) {
67 // create a random gate and add it to the circuit
68
69 // first, pick randomly three qubits (depending on the randomly chosen
70 // gate type, not all of them will be used)
71 Types::qubits_vector qubits(nrQubitsForRandomCirc);
72 std::iota(qubits.begin(), qubits.end(), 0);
73 std::shuffle(qubits.begin(), qubits.end(), g);
74 auto q1 = qubits[0];
75 auto q2 = qubits[1];
76
77 // now some random parameters, again, they might be ignored
78 const double param1 = *dblGenIter;
79 ++dblGenIter;
80
81 Circuits::QuantumGateType gateType =
82 static_cast<Circuits::QuantumGateType>(*gateGenIter);
83
84 auto theGate = Circuits::CircuitFactory<>::CreateGate(gateType, q1, q2, 0,
85 param1, 0, 0, 0);
86 if (!theGate->IsClifford()) {
87 --gateNr;
88 continue;
89 }
90
91 circ->AddOperation(theGate);
92 }
93 }
