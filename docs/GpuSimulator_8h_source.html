545 void ApplyCCX(Types::qubit_t qubit0, Types::qubit_t qubit1,
546 Types::qubit_t qubit2) override {
547 if (GetSimulationType() == SimulationType::kStatevector)
548 state->ApplyCCX(qubit0, qubit1, qubit2);
549 else if (GetSimulationType() == SimulationType::kMatrixProductState) {
550 const size_t q1 = qubit0; // control 1
551 const size_t q2 = qubit1; // control 2
552 const size_t q3 = qubit2; // target
553
554 // Sleator-Weinfurter decomposition
555 mps->ApplyCSX(static_cast<unsigned int>(q2),
556 static_cast<unsigned int>(q3));
557 mps->ApplyCX(static_cast<unsigned int>(q1),
558 static_cast<unsigned int>(q2));
559 mps->ApplyCSXDG(static_cast<unsigned int>(q2),
560 static_cast<unsigned int>(q3));
561 mps->ApplyCX(static_cast<unsigned int>(q1),
562 static_cast<unsigned int>(q2));
563 mps->ApplyCSX(static_cast<unsigned int>(q1),
564 static_cast<unsigned int>(q3));
565 } else if (GetSimulationType() == SimulationType::kTensorNetwork) {
566 tn->ApplyCCX(qubit0, qubit1, qubit2);
567 }
568
569 NotifyObservers({qubit2, qubit1, qubit0});
570 }

580 void ApplyCSwap(Types::qubit_t ctrl_qubit, Types::qubit_t qubit0,
581 Types::qubit_t qubit1) override {
582 if (GetSimulationType() == SimulationType::kStatevector)
583 state->ApplyCSwap(ctrl_qubit, qubit0, qubit1);
584 else if (GetSimulationType() == SimulationType::kMatrixProductState) {
585 const size_t q1 = ctrl_qubit; // control
586 const size_t q2 = qubit0;
587 const size_t q3 = qubit1;
588
589 // TODO: find a better decomposition
590 // this one I've got with the qiskit transpiler
591 mps->ApplyCX(static_cast<unsigned int>(q3),
592 static_cast<unsigned int>(q2));
593 mps->ApplyCSX(static_cast<unsigned int>(q2),
594 static_cast<unsigned int>(q3));
595 mps->ApplyCX(static_cast<unsigned int>(q1),
596 static_cast<unsigned int>(q2));
597
598 mps->ApplyP(static_cast<unsigned int>(q3), M_PI);
599 mps->ApplyP(static_cast<unsigned int>(q2), -M_PI_2);
600
601 mps->ApplyCSX(static_cast<unsigned int>(q2),
602 static_cast<unsigned int>(q3));
603 mps->ApplyCX(static_cast<unsigned int>(q1),
604 static_cast<unsigned int>(q2));
605
606 mps->ApplyP(static_cast<unsigned int>(q3), M_PI);
607
608 mps->ApplyCSX(static_cast<unsigned int>(q1),
609 static_cast<unsigned int>(q3));
610 mps->ApplyCX(static_cast<unsigned int>(q3),
611 static_cast<unsigned int>(q2));
612 } else if (GetSimulationType() == SimulationType::kTensorNetwork) {
613 tn->ApplyCSwap(ctrl_qubit, qubit0, qubit1);
614 }
615
616 NotifyObservers({qubit1, qubit0, ctrl_qubit});
617 }
