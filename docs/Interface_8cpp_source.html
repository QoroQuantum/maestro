120#ifdef _WIN32
121 __declspec(dllexport)
122#endif
123 char* SimpleExecute(unsigned long int simpleSim,
124 const char* circuitStr, const char* jsonConfig) {
125 if (simpleSim == 0 || !circuitStr || !jsonConfig || !maestroInstance)
126 return nullptr;
127
128 auto network = maestroInstance->GetSimpleSimulator(simpleSim);
129
130 // step 1: Parse the JSON circuit and configuration strings
131 // convert the JSON circuit into a Circuit object
132
133 // I'm unsure here on how it deals with the classical registers, more
134 // precisely with stuff like "other_measure_name" and "meas" (see below) since
135 // in the example it seems to just use the cbit number
136
137 // This is the json format:
138 // {"instructions":
139 // [{"name": "h", "qubits": [0], "params": []},
140 // {"name": "cx", "qubits": [0, 1], "params": []},
141 // {"name": "rx", "qubits": [0], "params": [0.39528385768119634]},
142 // {"name": "measure", "qubits": [0], "memory": [0]}],
143 //
144 // "num_qubits": 2, "num_clbits": 4,
145 // "quantum_registers": {"q": [0, 1]},
146 // "classical_registers": {"c": [0, 1], "other_measure_name": [2], "meas":
147 // [3]}}
148
149 std::shared_ptr<Circuits::Circuit<>> circuit;
150
151 if (circuitStr[0] == '{' || circuitStr[0] == '[') {
152 // assume JSON format only if either object or array
153 Json::JsonParserMaestro<> jsonParser;
154 circuit = jsonParser.ParseCircuit(circuitStr);
155 }
156 else {
157 // QASM 2.0 format
158 qasm::QasmToCirc<> parser;
159 std::string qasmInput(circuitStr);
160 circuit = parser.ParseAndTranslate(qasmInput);
161 if (parser.Failed())
162 return nullptr;
163 }
164
165 // check if the circuit has measurements only at the end
166
167 // get the number of shots from the configuration
168 size_t nrShots = 1; // default value
169
170 const auto configJson = Json::JsonParserMaestro<>::ParseString(jsonConfig);
171
172 if (configJson.is_object()) {
173 const auto configObject = configJson.as_object();
174 // get whatever else is needed from the configuration
175 // maybe simulator type, allowed simulator types, bond dimension limit, etc.
176
177 // execute the circuit in the network object
178 if (configObject.contains("shots") &&
179 configObject.at("shots").is_number()) {
180 auto number = configObject.at("shots");
181 nrShots = number.is_int64() ? (size_t)number.as_int64()
182 : (size_t)number.as_uint64();
183 }
184 }
185
186 bool configured = false;
187
188 const std::string maxBondDim = Json::JsonParserMaestro<>::GetConfigString(
189 "matrix_product_state_max_bond_dimension", configJson);
190 if (!maxBondDim.empty()) {
191 configured = true;
192 if (network->GetSimulator())
193 network->GetSimulator()->Clear();
194 network->Configure("matrix_product_state_max_bond_dimension",
195 maxBondDim.c_str());
196 }
197
198 const std::string singularValueThreshold =
199 Json::JsonParserMaestro<>::GetConfigString(
200 "matrix_product_state_truncation_threshold", configJson);
201 if (!singularValueThreshold.empty()) {
202 configured = true;
203 if (network->GetSimulator())
204 network->GetSimulator()->Clear();
205 network->Configure("matrix_product_state_truncation_threshold",
206 singularValueThreshold.c_str());
207 }
208
209 const std::string mpsSample = Json::JsonParserMaestro<>::GetConfigString(
210 "mps_sample_measure_algorithm", configJson);
211 if (!mpsSample.empty()) {
212 configured = true;
213 if (network->GetSimulator())
214 network->GetSimulator()->Clear();
215 network->Configure("mps_sample_measure_algorithm", mpsSample.c_str());
216 }
217
218 if (configured || !network->GetSimulator())
219 network->CreateSimulator();
220
221 // TODO: get from config the allowed simulators types and so on, if set
222 auto start = std::chrono::high_resolution_clock::now();
223 auto results = network->RepeatedExecuteOnHost(circuit, 0, nrShots);
224 auto end = std::chrono::high_resolution_clock::now();
225
226 std::chrono::duration<double> duration = end - start;
227 double time_taken = duration.count();
228 std::string timeStr = std::to_string(time_taken);
229
230 // convert the results into a JSON string
231 // allocate memory for the result string and copy the JSON result into it
232 // return the result string
233
234 boost::json::object jsonResult;
235 jsonResult.reserve(results.size());
236
237 for (auto& result : results) {
238 boost::json::string bits;
239 bits.reserve(result.first.size());
240 for (const auto bit : result.first)
241 bits.append(bit ? "1" : "0");
242
243 jsonResult.emplace(std::move(bits), std::move(result.second));
244 }
245
246 boost::json::object response;
247 response.reserve(4);
248
249 response.emplace("counts", std::move(jsonResult));
250 response.emplace("time_taken", timeStr);
251
252 auto simulatorType = network->GetLastSimulatorType();
253
254 switch (simulatorType) {
255#ifndef NO_QISKIT_AER
256 case Simulators::SimulatorType::kQiskitAer:
257 response.emplace("simulator", "aer");
258 break;
259#endif
260 case Simulators::SimulatorType::kQCSim:
261 response.emplace("simulator", "qcsim");
262 break;
263#ifndef NO_QISKIT_AER
264 case Simulators::SimulatorType::kCompositeQiskitAer:
265 response.emplace("simulator", "composite_aer");
266 break;
267#endif
268 case Simulators::SimulatorType::kCompositeQCSim:
269 response.emplace("simulator", "composite_qcsim");
270 break;
271#ifdef __linux__
272 case Simulators::SimulatorType::kGpuSim:
273 response.emplace("simulator", "gpu_simulator");
274 break;
275#endif
276 default:
277 response.emplace("simulator", "unknown");
278 break;
279 }
280
281 auto simulationType = network->GetLastSimulationType();
282 switch (simulationType) {
283 case Simulators::SimulationType::kStatevector:
284 response.emplace("method", "statevector");
285 break;
286 case Simulators::SimulationType::kMatrixProductState:
287 response.emplace("method", "matrix_product_state");
288 break;
289 case Simulators::SimulationType::kStabilizer:
290 response.emplace("method", "stabilizer");
291 break;
292 case Simulators::SimulationType::kTensorNetwork:
293 response.emplace("method", "tensor_network");
294 break;
295 default:
296 response.emplace("method", "unknown");
297 break;
298 }
299
300 const std::string responseStr = boost::json::serialize(response);
301 const size_t responseSize = responseStr.length();
302 char* result = new char[responseSize + 1];
303
304 const char* responseData = responseStr.c_str();
305 std::copy(responseData, responseData + responseSize, result);
306
307 result[responseSize] = 0; // ensure null-termination
308
309 return result;
310 }
