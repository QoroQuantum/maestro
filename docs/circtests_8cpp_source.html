172 // fills randomly the circuit with gates
173 void GenerateCircuit(int nrGates, int nrQubits) {
174 std::random_device rd;
175 std::mt19937 g(rd());
176
177 auto dblGen = bdata::random(-2. * M_PI, 2. * M_PI);
178 auto dblGenIter = dblGen.begin();
179
180 auto gateGen = bdata::random(
181 0, static_cast<int>(Circuits::QuantumGateType::kCCXGateType));
182 auto gateGenIter = gateGen.begin();
183
184 // TODO: Maybe insert from time to time a random number generating 'gate'
185 // and a conditioned random one, those should not affect results?
186 for (int gateNr = 0; gateNr < nrGates; ++gateNr, ++gateGenIter) {
187 // create a random gate and add it to the circuit
188
189 // first, pick randomly three qubits (depending on the randomly chosen
190 // gate type, not all of them will be used)
191 Types::qubits_vector qubits(nrQubits);
192 std::iota(qubits.begin(), qubits.end(), 0);
193 std::shuffle(qubits.begin(), qubits.end(), g);
194 auto q1 = qubits[0];
195 auto q2 = qubits[1];
196 auto q3 = qubits[2];
197
198 // now some random parameters, again, they might be ignored
199 const double param1 = *dblGenIter;
200 ++dblGenIter;
201 const double param2 = *dblGenIter;
202 ++dblGenIter;
203 const double param3 = *dblGenIter;
204 ++dblGenIter;
205 const double param4 = *dblGenIter;
206 ++dblGenIter;
207
208 const Circuits::QuantumGateType gateType =
209 static_cast<Circuits::QuantumGateType>(*gateGenIter);
210
211 auto theGate = Circuits::CircuitFactory<>::CreateGate(
212 gateType, q1, q2, q3, param1, param2, param3, param4);
213 randomCirc->AddOperation(theGate);
214 }
215 }
