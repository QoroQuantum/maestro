203 void Configure(const char *key, const char *value) override {
204 if (std::string("method") == key) {
205 if (std::string("statevector") == value)
206 simulationType = SimulationType::kStatevector;
207 else if (std::string("matrix_product_state") == value)
208 simulationType = SimulationType::kMatrixProductState;
209 } else if (std::string("matrix_product_state_truncation_threshold") ==
210 key) {
211 singularValueThreshold = std::stod(value);
212 if (singularValueThreshold > 0.) {
213 limitEntanglement = true;
214 if (mps) mps->SetCutoff(singularValueThreshold);
215 } else
216 limitEntanglement = false;
217 } else if (std::string("matrix_product_state_max_bond_dimension") == key) {
218 chi = std::stoi(value);
219 if (chi > 0) {
220 limitSize = true;
221 if (mps) mps->SetMaxExtent(chi);
222 } else
223 limitSize = false;
224 }

300 size_t Measure(const Types::qubits_vector &qubits) override {
301 // TODO: this is inefficient, maybe implement it better in gpu sim

305 size_t res = 0;
306 size_t mask = 1ULL;

308 DontNotify();
309 if (simulationType == SimulationType::kStatevector) {

312 if (state->MeasureQubitCollapse(static_cast<int>(qubit))) res |= mask;

335 void ApplyReset(const Types::qubits_vector &qubits) override {
336 DontNotify();
337 if (simulationType == SimulationType::kStatevector) {
338 for (size_t qubit : qubits)
339 if (state->MeasureQubitCollapse(static_cast<int>(qubit)))
340 state->ApplyX(static_cast<int>(qubit));
341 } else if (simulationType == SimulationType::kMatrixProductState) {
342 for (size_t qubit : qubits)
343 if (mps->Measure(static_cast<unsigned int>(qubit)))
344 mps->ApplyX(static_cast<unsigned int>(qubit));
345 }
346
347 Notify();
348 NotifyObservers(qubits);

383 std::complex<double> Amplitude(Types::qubit_t outcome) override {
384 double real = 0.0;
385 double imag = 0.0;
386
387 if (simulationType == SimulationType::kStatevector)
388 state->Amplitude(outcome, &real, &imag);
389 else if (simulationType == SimulationType::kMatrixProductState) {
390 std::vector<long int> fixedValues(nrQubits);
391 for (size_t i = 0; i < nrQubits; ++i)
392 fixedValues[i] = (outcome & (1ULL << i)) ? 1 : 0;
393 mps->Amplitude(nrQubits, fixedValues.data(), &real, &imag);
394 }
395
396 return std::complex<double>(real, imag);

409 std::vector<double> AllProbabilities() override {
410 if (nrQubits == 0) return {};
411 const size_t numStates = 1ULL << nrQubits;
412 std::vector<double> result(numStates);
413
414 if (simulationType == SimulationType::kStatevector)
415 state->AllProbabilities(result.data());
416 else if (simulationType == SimulationType::kMatrixProductState) {

468 std::unordered_map<Types::qubit_t, Types::qubit_t> SampleCounts(
469 const Types::qubits_vector &qubits, size_t shots = 1000) override {
470 if (qubits.empty() || shots == 0) return {};
471
472 std::unordered_map<Types::qubit_t, Types::qubit_t> result;
473
474 DontNotify();
475
476 if (simulationType == SimulationType::kStatevector) {
477 std::vector<long int> samples(shots);
478 state->SampleAll(shots, samples.data());
479
480 for (auto outcome : samples) ++result[outcome];
481 } else if (simulationType == SimulationType::kMatrixProductState) {
482 std::unordered_map<std::vector<bool>, int64_t> *map =
483 mps->GetMapForSample();
484
485 std::vector<unsigned int> qubitsIndices(qubits.begin(), qubits.end());
486
487 mps->Sample(shots, qubitsIndices.size(), qubitsIndices.data(), map);
488
489 // put the results in the result map
490 for (const auto &[meas, cnt] : *map) {
491 Types::qubit_t outcome = 0;
492 Types::qubit_t mask = 1ULL;
493 for (Types::qubit_t q = 0; q < qubits.size(); ++q) {
494 if (meas[q]) outcome |= mask;
495 mask <<= 1;
496 }
497
498 result[outcome] += cnt;
499 }
500
501 mps->FreeMapForSample(map);
502 }
503
504 Notify();
505 NotifyObservers(qubits);
506
507 return result;
508 }

681 Types::qubit_t MeasureNoCollapse() override {
682 if (simulationType == SimulationType::kStatevector)
683 return state->MeasureAllQubitsNoCollapse();
684 else if (simulationType == SimulationType::kMatrixProductState) {
685 Types::qubits_vector fixedValues(nrQubits);
686 std::iota(fixedValues.begin(), fixedValues.end(), 0);
687 const auto res = SampleCounts(fixedValues, 1);
