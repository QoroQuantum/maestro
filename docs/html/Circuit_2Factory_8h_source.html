121 static const std::shared_ptr<IQuantumGate<Time>>
122 CreateGate(QuantumGateType type, size_t q1, size_t q2 = 0, size_t q3 = 0,
123 double param1 = 0, double param2 = 0, double param3 = 0,
124 double param4 = 0) {
125 // parameters that do not make sense for a gate are ignored, eg if it's a
126 // single qubit gate, q2 and q3 are ignored if it doesn't have a parameter,
127 // the parameter is ignored if it has only one, the others are ignored
128 // TODO: maybe make specialized variants, as in CreateOneQubitGate,
129 // CreateTwoQubitGate, etc
130 std::shared_ptr<IQuantumGate<Time>> gate;
131
132 switch (type) {
133 // one qubit gates
134 case QuantumGateType::kPhaseGateType:
135 gate = std::make_shared<PhaseGate<Time>>(q1, param1);
136 break;
137 case QuantumGateType::kXGateType:
138 gate = std::make_shared<XGate<Time>>(q1);
139 break;
140 case QuantumGateType::kYGateType:
141 gate = std::make_shared<YGate<Time>>(q1);
142 break;
143 case QuantumGateType::kZGateType:
144 gate = std::make_shared<ZGate<Time>>(q1);
145 break;
146 case QuantumGateType::kHadamardGateType:
147 gate = std::make_shared<HadamardGate<Time>>(q1);
148 break;
149 case QuantumGateType::kSGateType:
150 gate = std::make_shared<SGate<Time>>(q1);
151 break;
152 case QuantumGateType::kSdgGateType:
153 gate = std::make_shared<SdgGate<Time>>(q1);
154 break;
155 case QuantumGateType::kTGateType:
156 gate = std::make_shared<TGate<Time>>(q1);
157 break;
158 case QuantumGateType::kTdgGateType:
159 gate = std::make_shared<TdgGate<Time>>(q1);
160 break;
161 case QuantumGateType::kSxGateType:
162 gate = std::make_shared<SxGate<Time>>(q1);
163 break;
164 case QuantumGateType::kSxDagGateType:
165 gate = std::make_shared<SxDagGate<Time>>(q1);
166 break;
167 case QuantumGateType::kKGateType:
168 gate = std::make_shared<KGate<Time>>(q1);
169 break;
170 case QuantumGateType::kRxGateType:
171 gate = std::make_shared<RxGate<Time>>(q1, param1);
172 break;
173 case QuantumGateType::kRyGateType:
174 gate = std::make_shared<RyGate<Time>>(q1, param1);
175 break;
176 case QuantumGateType::kRzGateType:
177 gate = std::make_shared<RzGate<Time>>(q1, param1);
178 break;
179 case QuantumGateType::kUGateType:
180 gate = std::make_shared<UGate<Time>>(q1, param1, param2, param3, param4);
181 break;
182 // two qubit gates
183 case QuantumGateType::kSwapGateType:
184 gate = std::make_shared<SwapGate<Time>>(q1, q2);
185 break;
186 case QuantumGateType::kCXGateType:
187 gate = std::make_shared<CXGate<Time>>(q1, q2);
188 break;
189 case QuantumGateType::kCYGateType:
190 gate = std::make_shared<CYGate<Time>>(q1, q2);
191 break;
192 case QuantumGateType::kCZGateType:
193 gate = std::make_shared<CZGate<Time>>(q1, q2);
194 break;
195 case QuantumGateType::kCPGateType:
196 gate = std::make_shared<CPGate<Time>>(q1, q2, param1);
197 break;
198 case QuantumGateType::kCRxGateType:
199 gate = std::make_shared<CRxGate<Time>>(q1, q2, param1);
200 break;
201 case QuantumGateType::kCRyGateType:
202 gate = std::make_shared<CRyGate<Time>>(q1, q2, param1);
203 break;
204 case QuantumGateType::kCRzGateType:
205 gate = std::make_shared<CRzGate<Time>>(q1, q2, param1);
206 break;
207 case QuantumGateType::kCHGateType:
208 gate = std::make_shared<CHGate<Time>>(q1, q2);
209 break;
210 case QuantumGateType::kCSxGateType:
211 gate = std::make_shared<CSxGate<Time>>(q1, q2);
212 break;
213 case QuantumGateType::kCSxDagGateType:
214 gate = std::make_shared<CSxDagGate<Time>>(q1, q2);
215 break;
216 case QuantumGateType::kCUGateType:
217 gate = std::make_shared<CUGate<Time>>(q1, q2, param1, param2, param3,
218 param4);
219 break;
220 // three qubit gates
221 case QuantumGateType::kCSwapGateType:
222 gate = std::make_shared<CSwapGate<Time>>(q1, q2, q3);
223 break;
224 case QuantumGateType::kCCXGateType:
225 gate = std::make_shared<CCXGate<Time>>(q1, q2, q3);
226 break;
227 default:
228 break;
229 }
230
231 return gate;
232 }

464 static std::vector<std::shared_ptr<IOperation<Time>>>
465 CreateTeleportationCircuit(size_t entqbit1, size_t entqbit2, size_t srcqbit,
466 size_t cbit1, size_t cbit2) {
467 std::vector<std::shared_ptr<IOperation<Time>>> ops(8);
468
469 // EPR pair
470 ops[0] = CreateGate(QuantumGateType::kHadamardGateType, entqbit1);
471 ops[1] = CreateGate(QuantumGateType::kCXGateType, entqbit1, entqbit2);
472
473 // teleportation follows, using the entangled qubits
474 ops[2] = CreateGate(QuantumGateType::kCXGateType, srcqbit, entqbit1);
475 ops[3] = CreateGate(QuantumGateType::kHadamardGateType, srcqbit);
476
477 // measurements
478 ops[4] = CreateMeasurement({{srcqbit, cbit1}});
479 ops[5] = CreateMeasurement({{entqbit1, cbit2}});
480
481 // apply the conditional gates based on the measurements
482 const auto xgate = CreateGate(QuantumGateType::kXGateType, entqbit2);
483 const auto zgate = CreateGate(QuantumGateType::kZGateType, entqbit2);
484
485 ops[6] = CreateSimpleConditionalGate(xgate, cbit2);
486 ops[7] = CreateSimpleConditionalGate(zgate, cbit1);
487
488 return ops;
489 }

503 static std::vector<std::shared_ptr<IOperation<Time>>>
504 CreateStartDistributionCircuit(size_t ctrlQubit, size_t ctrlEntangledQubit,
505 size_t tgtEntangledQubit,
506 size_t ctrlEntangledMeasureBit) {
507 std::vector<std::shared_ptr<IOperation<Time>>> ops(5);
508 // 1. Entangle qubits for the two hosts
509 // auto ops = CreateBellStateCircuit(ctrlEntangledQubit, tgtEntangledQubit);
510 ops[0] = CreateGate(QuantumGateType::kHadamardGateType, ctrlEntangledQubit);
511 ops[1] = CreateGate(QuantumGateType::kCXGateType, ctrlEntangledQubit,
512 tgtEntangledQubit);
513
514 // 2. Now apply cnot from ctrl to entangled qubit on ctrl host
515 // ops.emplace_back(std::make_shared<Circuits::CXGate<Time>>(ctrlQubit,
516 // ctrlEntangledQubit));
517 ops[2] =
518 CreateGate(QuantumGateType::kCXGateType, ctrlQubit, ctrlEntangledQubit);
519 // 3. Measure the entangled qubit on ctrl host (then the measurement result
520 // is sent to the other host)
521 const std::vector<std::pair<Types::qubit_t, size_t>> measureOps{
522 {ctrlEntangledQubit, ctrlEntangledMeasureBit}};
523 // ops.emplace_back(std::make_shared<Circuits::MeasurementOperation<Time>>(measureOps));
524 ops[3] = CreateMeasurement(measureOps);
525
526 // 4. Send the measurement result to the target host, use the result to
527 // apply not on the target entangled qubit
528 // ops.emplace_back(std::make_shared<Circuits::ConditionalGate<Time>>(std::make_shared<Circuits::XGate<Time>>(tgtEntangledQubit),
529 // std::make_shared<Circuits::EqualCondition>(std::vector<size_t>{
530 // ctrlEntangledMeasureBit }, std::vector<bool>{true})));
531 ops[4] = std::make_shared<Circuits::ConditionalGate<Time>>(
532 std::make_shared<Circuits::XGate<Time>>(tgtEntangledQubit),
533 std::make_shared<Circuits::EqualCondition>(
534 std::vector<size_t>{ctrlEntangledMeasureBit},
535 std::vector<bool>{true}));
536
537 return ops;
538 }

551 static std::vector<std::shared_ptr<IOperation<Time>>>
552 CreateEndDistributionCircuit(size_t ctrlQubit, size_t tgtEntangledQubit,
553 size_t tgtEntangledMeasureBit) {
554 std::vector<std::shared_ptr<IOperation<Time>>> ops(3);
555
556 ops[0] = std::make_shared<Circuits::HadamardGate<Time>>(tgtEntangledQubit);
557 const std::vector<std::pair<Types::qubit_t, size_t>> measureOps{
558 {tgtEntangledQubit, tgtEntangledMeasureBit}};
559 ops[1] = std::make_shared<Circuits::MeasurementOperation<Time>>(measureOps);
560 ops[2] = std::make_shared<Circuits::ConditionalGate<Time>>(
561 std::make_shared<Circuits::ZGate<Time>>(ctrlQubit),
562 std::make_shared<Circuits::EqualCondition>(
563 std::vector<size_t>{tgtEntangledMeasureBit},
564 std::vector<bool>{true}));

566 return ops;
